```dataviewjs
try {
    const folder = "Logistics/购物/转运记录";
    const pages = dv.pages(`"${folder}"`).where(p => p.file.name !== "转运记录" && p.打包时间);
    const today = dv.date("today");
    const monthLabels = ["1月","2月","3月","4月","5月","6月","7月","8月","9月","10月","11月","12月"];

    const records = [];
    for (const p of pages) {
        const startDate = p.打包时间;
        let endDate = (startDate > today) ? startDate : (p.收到时间 || today);
        const total = (p["商品金额($)"] || 0) + (p["运费($)"] || 0);
        if (startDate && total > 0) {
            records.push({ start: new Date(startDate.ts), end: new Date(endDate.ts), total });
        }
    }

    if (records.length === 0) {
        dv.paragraph("No data.");
    } else {
        const allYears = [...new Set(records.map(r => r.start.getFullYear()))].sort();
        const colors = ['#60a5fa','#4ade80','#fb923c','#e879f9','#fbbf24'];

        // Calculate monthly totals per year
        const yearData = {};
        for (const year of allYears) {
            yearData[year] = monthLabels.map((_, mi) => {
                const monthStart = new Date(year, mi, 1);
                const monthEnd   = new Date(year, mi + 1, 0);
                let total = 0;
                for (const r of records) {
                    const orderDays    = Math.ceil((r.end - r.start) / 86400000) + 1;
                    const dailyRate    = r.total / orderDays;
                    const overlapStart = new Date(Math.max(r.start, monthStart));
                    const overlapEnd   = new Date(Math.min(r.end,   monthEnd));
                    if (overlapStart <= overlapEnd) {
                        const overlapDays = Math.ceil((overlapEnd - overlapStart) / 86400000) + 1;
                        total += dailyRate * overlapDays;
                    }
                }
                return total > 0 ? parseFloat(total.toFixed(2)) : 0;
            });
        }

        // Chart dimensions
        const W = 700, H = 300;
        const pad = { top: 40, right: 20, bottom: 50, left: 60 };
        const plotW = W - pad.left - pad.right;
        const plotH = H - pad.top - pad.bottom;
        const nYears = allYears.length;
        const groupW = plotW / 12;
        const barW = Math.max(4, (groupW - 4) / nYears);

        const allValues = Object.values(yearData).flat().filter(v => v > 0);
        const maxVal = allValues.length > 0 ? Math.max(...allValues) : 10;
        const yAxisMax = Math.ceil(maxVal * 1.15 / 10) * 10 || 10;
        const ySteps = 5;

        // Y axis grid
        let gridSVG = '';
        for (let i = 0; i <= ySteps; i++) {
            const val = (yAxisMax / ySteps) * i;
            const y = pad.top + plotH - (val / yAxisMax) * plotH;
            gridSVG += `<text x="${pad.left - 6}" y="${y + 4}" text-anchor="end" font-size="10" fill="#888">$${val.toFixed(0)}</text>`;
            gridSVG += `<line x1="${pad.left}" y1="${y}" x2="${pad.left + plotW}" y2="${y}" stroke="#ccc" stroke-dasharray="2,3"/>`;
        }

        // Bars
        let barsSVG = '';
        for (let mi = 0; mi < 12; mi++) {
            const groupX = pad.left + mi * groupW;
            for (let yi = 0; yi < nYears; yi++) {
                const val = yearData[allYears[yi]][mi];
                if (val <= 0) continue;
                const barH = (val / yAxisMax) * plotH;
                const x = groupX + yi * barW + 2;
                const y = pad.top + plotH - barH;
                const color = colors[yi % colors.length];
                barsSVG += `<rect x="${x.toFixed(1)}" y="${y.toFixed(1)}" width="${barW.toFixed(1)}" height="${barH.toFixed(1)}" fill="${color}" opacity="0.85" rx="2"/>`;
                if (barH > 14) {
                    barsSVG += `<text x="${(x + barW/2).toFixed(1)}" y="${(y - 2).toFixed(1)}" text-anchor="middle" font-size="8" fill="${color}">$${val.toFixed(0)}</text>`;
                }
            }
        }

        // X axis labels
        let xLabelSVG = '';
        for (let mi = 0; mi < 12; mi++) {
            const cx = pad.left + mi * groupW + groupW / 2;
            xLabelSVG += `<text x="${cx.toFixed(1)}" y="${pad.top + plotH + 16}" text-anchor="middle" font-size="10" fill="#888">${monthLabels[mi]}</text>`;
        }

        // Legend
        let legendSVG = '';
        for (let yi = 0; yi < nYears; yi++) {
            const lx = pad.left + yi * 70;
            const ly = H - 10;
            legendSVG += `<rect x="${lx}" y="${ly - 8}" width="10" height="10" fill="${colors[yi % colors.length]}" rx="2"/>`;
            legendSVG += `<text x="${lx + 14}" y="${ly}" font-size="10" fill="#888">${allYears[yi]}年</text>`;
        }

        const svg = `<div style="overflow-x:auto; padding:8px 0;">
  <svg width="${W}" height="${H}" style="font-family:var(--font-interface);">
    <text x="${W/2}" y="18" text-anchor="middle" font-size="13" font-weight="600" fill="var(--text-normal)">转运月花费 ($)</text>
    <text x="14" y="${pad.top + plotH/2}" text-anchor="middle" font-size="10" fill="#888" transform="rotate(-90,14,${pad.top + plotH/2})">$</text>
    ${gridSVG}
    <line x1="${pad.left}" y1="${pad.top}" x2="${pad.left}" y2="${pad.top + plotH}" stroke="#999"/>
    <line x1="${pad.left}" y1="${pad.top + plotH}" x2="${pad.left + plotW}" y2="${pad.top + plotH}" stroke="#999"/>
    ${barsSVG}
    ${xLabelSVG}
    ${legendSVG}
  </svg>
</div>`;

        dv.paragraph(svg);
    }
} catch (e) {
    dv.paragraph('⚠️ Error: ' + e.message);
}
```

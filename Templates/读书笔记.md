try {
    const page = dv.current();
    const current = page?.完成页数;
    const total = page?.totalPage;
    const currentNum = typeof current === "number" ? current : (typeof current === "string" ? parseFloat(current) : 0);
    const totalNum = typeof total === "number" ? total : (typeof total === "string" ? parseFloat(total) : 0);

    if (isNaN(currentNum) || isNaN(totalNum) || totalNum <= 0) {
        dv.container.innerHTML = `—`;
    } else {
        const pct = Math.min(100, Math.max(0, (currentNum / totalNum) * 100));
        const W = Math.max(160, Math.min(500, (dv.container.clientWidth || 320) - 8));
        const H = 18;
        const R = 9;
        const filled = (pct / 100) * W;

        const svg = `
        <svg width="${W}" height="${H}" style="vertical-align:middle;">
          <defs>
            <linearGradient id="pg" x1="0" x2="1" y1="0" y2="0">
              <stop offset="0%" stop-color="#4ade80"/>
              <stop offset="100%" stop-color="#16a34a"/>
            </linearGradient>
            <clipPath id="rr">
              <rect width="${W}" height="${H}" rx="${R}"/>
            </clipPath>
          </defs>
          <rect width="${W}" height="${H}" rx="${R}" fill="#e5e7eb"/>
          <rect width="${filled}" height="${H}" fill="url(#pg)" clip-path="url(#rr)"/>
        </svg>
        <span style="margin-left:8px;font-size:1.1em;">${pct.toFixed(2)}%</span>`;

        dv.container.innerHTML = svg;
    }
} catch (error) {
    dv.container.innerHTML = `⚠️ 错误`;
}

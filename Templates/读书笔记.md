```dataviewjs
try {
    const page = dv.current();
    const current = page?.完成页数;
    const total = page?.totalPage;
    const currentNum = typeof current === "number" ? current : (typeof current === "string" ? parseFloat(current) : 0);
    const totalNum = typeof total === "number" ? total : (typeof total === "string" ? parseFloat(total) : 0);

    if (isNaN(currentNum) || isNaN(totalNum) || totalNum <= 0) {
        dv.container.innerHTML = `—`;
    } else {
        const percentage = Math.min(100, (currentNum / totalNum) * 100);
        const blockCount = 30;
        const moons = ["🌑", "🌘", "🌗", "🌖", "🌕"];
        const totalUnits = blockCount * (moons.length - 1);
        const filledUnits = Math.max(0, Math.min(totalUnits, Math.round((percentage / 100) * totalUnits)));
        let moonBar = "";
        for (let i = 0; i < blockCount; i++) {
            const unitValue = filledUnits - i * (moons.length - 1);
            const moonIndex = Math.max(0, Math.min(moons.length - 1, unitValue));
            moonBar += moons[moonIndex];
        }
        dv.container.style.cssText = "display:inline;white-space:nowrap;";
        dv.container.innerHTML = `${moonBar} ${percentage.toFixed(2)}%`;
    }
} catch (error) {
    dv.container.innerHTML = `⚠️ 错误`;
}
```
